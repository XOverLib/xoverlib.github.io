// ===== БРОНИРОВАНИЕ ЧЕРЕЗ GITHUB ISSUES ===== // Токен для GitHub API (создай свой с правами repo:public_repo) // Храни токен в отдельном файле или переменной окружения const GITHUB_TOKEN = 'ghp_ваш_токен'; // ЗАМЕНИ НА РЕАЛЬНЫЙ ТОКЕН const GITHUB_REPO = 'ваш-аккаунт/XOver'; // ЗАМЕНИ НА СВОЙ РЕПОЗИТОРИЙ async function openReservation(item) { const bookTitle = item['Название'] || 'Без названия'; const bookId = item['ID'] || item['Название'] || 'unknown'; // Спрашиваем QR-код или ID клиента const clientInput = prompt('Введите ID клиента или отсканируйте QR-код:'); if (!clientInput) return; // Парсим клиента (если QR-код в формате "ID|имя|...") let clientId = clientInput; if (clientInput.includes('|')) { const parts = clientInput.split('|'); clientId = parts[0] || clientInput; } // Создаём заявку в GitHub Issues try { const response = await fetch(`https://api.github.com/repos/${GITHUB_REPO}/issues`, { method: 'POST', headers: { 'Authorization': `token ${GITHUB_TOKEN}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ title: `📚 Бронирование: ${bookTitle}`, body: `## Заявка на бронирование\n\n` + `**Книга:** ${bookTitle}\n` + `**Клиент ID:** ${clientId}\n` + `**Дата:** ${new Date().toLocaleString()}\n\n` + `---\n` + `_Для обработки заявки откройте админ-панель._`, labels: ['reservation', 'pending'] }) }); if (response.ok) { const issue = await response.json(); alert(`✅ Книга забронирована!\nНомер заявки: #${issue.number}\nАдминистратор скоро её обработает.`); } else { const error = await response.json(); console.error('GitHub API error:', error); alert('❌ Ошибка бронирования. Попробуйте позже.'); } } catch (error) { console.error('Network error:', error); alert('❌ Ошибка соединения. Проверьте интернет.'); } }
